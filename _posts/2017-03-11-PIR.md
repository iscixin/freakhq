在模塊的底部，有二個可變電阻，標示為 SX 及 TX。
SX 是調整靈敏度，順時針轉動則靈敏度調高，反之則低。
TX 是調整延時的時間長短，順時針轉動則延時加長，反之則減短。

腳位很簡單，只有三支腳：VCC(5V電源)、OUT(訊號)、GND(接地)。

## 範例程式

{% highlight c %}
const int PIRPIN=7;

void setup()
{
  pinMode(PIRPIN , INPUT);
  Serial.begin(9600);
}

void loop()
{
  int val=digitalRead(PIRPIN);
  if (val==HIGH) {
    Serial.println(val);
    //若有人體通過偵測到時
  } else {
    Serial.println(val);
    //沒有偵測到時
  }
}

{% endhighlight %}
